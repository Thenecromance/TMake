cmake_minimum_required(VERSION 3.12)

set(SHOW_LOG TRUE)
message("Import TLog module")
macro(BASIC_INFO)
    if(${SHOW_LOG})
        LOG_INFO(${CMAKE_VERSION})
    endif()
endmacro()

macro(LOG_INFO text)
    if(${SHOW_LOG})
        string(TIMESTAMP cur_time "%H:%M:%S")
        message("[TMAKE][INFO][${cur_time}]:" ${text})
    endif()
endmacro(LOG_INFO text)

macro(LOG_ERROR message)
    if(${SHOW_LOG})
        EMPTY_LINE()
        string(TIMESTAMP cur_time "%H:%M:%S")
        message(FATAL_ERROR "[TMAKE][INFO][${cur_time}]:" ${text})
        EMPTY_LINE()
    endif()
endmacro()

macro(EMPTY_LINE)
    if(${SHOW_LOG})
        message("")
    endif()
endmacro()
